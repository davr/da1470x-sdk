673 __STATIC_FORCEINLINE void hw_clk_set_sysclk(sys_clk_is_t mode)
674 {
675  /* Make sure a valid sys clock is requested */
676  ASSERT_WARNING(mode <= SYS_CLK_IS_PLL);
677 
678  /* Switch to PLL is only allowed when current system clock is XTAL32M */
679  ASSERT_WARNING(mode != SYS_CLK_IS_PLL ||
680  REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_XTAL32M) ||
681  REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL))
682 
683  /* Switch from PLL is only allowed when new system clock is XTAL32M */
684  ASSERT_WARNING(!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL) ||
685  mode == SYS_CLK_IS_XTAL32M ||
686  mode == SYS_CLK_IS_PLL);
687 
688  if (mode == SYS_CLK_IS_XTAL32M && REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RCHS)) {
689  ASSERT_WARNING(hw_clk_check_xtalm_status());
690 
691  GLOBAL_INT_DISABLE();
692  REG_SET_BIT(CRG_TOP, CLK_SWITCH2XTAL_REG, SWITCH2XTAL);
693  GLOBAL_INT_RESTORE();
694  } else {
695  GLOBAL_INT_DISABLE();
696  REG_SETF(CRG_TOP, CLK_CTRL_REG, SYS_CLK_SEL, mode);
697  GLOBAL_INT_RESTORE();
698  }
699 
700  /* Wait until the switch is done! */
701  switch (mode) {
702  case SYS_CLK_IS_XTAL32M:
703  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_XTAL32M)) {
704  }
705  return;
706 
707  case SYS_CLK_IS_RCHS:
708  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RCHS)) {
709  }
710  return;
711 
712  case SYS_CLK_IS_RCLP:
713  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_RCLP)) {
714  }
715  return;
716 
717  case SYS_CLK_IS_PLL:
718  while (!REG_GETF(CRG_TOP, CLK_CTRL_REG, RUNNING_AT_PLL)) {
719  }
720  return;
721  default:
722  ASSERT_WARNING(0);
723  }
724 }
