518 __STATIC_INLINE uint32_t hw_gpadc_internal_scaler_compensate(HW_GPADC_INPUT_POSITIVE channel, uint16_t adc_val)
519 {
520  if (channel == HW_GPADC_INP_VSYS) {
521  /* Scaler 0.157 x VSYS */
522  return (adc_val * 51) / 8;
523  } else if (channel == HW_GPADC_INP_VBUS) {
524  /* Scaler 0.164 x VBUS */
525  return (adc_val * 214) / 35;
526  } else if (channel == HW_GPADC_INP_VBAT) {
527  /* Scaler 0.189 x VBAT */
528  return (adc_val * 37) / 7;
529  }
530  return adc_val;
531 }
