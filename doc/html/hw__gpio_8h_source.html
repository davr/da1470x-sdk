530 {
531  GLOBAL_INT_DISABLE();
532  uint32_t p0_etm_pins = 0;
533  uint32_t p1_etm_pins = 0;
534 
535  #if (MAIN_PROCESSOR_BUILD)
536  if (REG_GETF(GPREG, DEBUG_REG, ETM_TRACE_MAP_ON_PINS_EN)) {
537  p0_etm_pins |= BIT31; /* TRACE_CLK */
538  p1_etm_pins |= BIT30; /* TRACE_DATA[0] */
539 
540  /* Get trace port size from Current Parallel Port Size Register (CSPSR) */
541  uint32_t trace_port_width = TPI->CSPSR;
542  if (trace_port_width > 1) {
543  p1_etm_pins |= BIT22; /* TRACE_DATA[1] */
544  }
545  if (trace_port_width == 4) {
546  p1_etm_pins |= BIT31; /* TRACE_DATA[2] */
547  p1_etm_pins |= BIT23; /* TRACE_DATA[3] */
548  }
549  }
550  #endif

552  REG_SETF(CRG_TOP, P0_RESET_PAD_LATCH_REG, P0_RESET_LATCH_EN,
553  CRG_TOP_P0_RESET_PAD_LATCH_REG_P0_RESET_LATCH_EN_Msk & ~p0_etm_pins);
554  REG_SETF(CRG_TOP, P1_RESET_PAD_LATCH_REG, P1_RESET_LATCH_EN,
555  CRG_TOP_P1_RESET_PAD_LATCH_REG_P1_RESET_LATCH_EN_Msk & ~p1_etm_pins);
