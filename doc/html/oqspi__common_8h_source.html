77 /*
78  * Some Octa FLash memories (e.g. Macronix) use dual byte opcode where the second byte equals to the
79  * inverted value of the first one. The same memories use single byte opcode when configured in single
80  * bus mode. The macro USE_DUAL_BYTE_OPCODE() checks if the dual byte must be used. If this is the case,
81  * the CONVERT_OPCODE_TO_DUAL_BYTE() has to be used to convert the single byte opcode to dual byte opcode.
82  */
83 #define USE_DUAL_BYTE_OPCODE(opcode_len, bus_mode) ((opcode_len == HW_OQSPI_OPCODE_LEN_2_BYTES) && \
84  (bus_mode == HW_OQSPI_BUS_MODE_OCTA))
85 #define CONVERT_OPCODE_TO_DUAL_BYTE(opcode) ((((uint16_t) (opcode) << 8) & 0xFF00) | (~(opcode) & 0x00FF))
