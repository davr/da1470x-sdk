114 __STATIC_INLINE void hw_dcache_write_flush(void)
115 {
116  REG_SET_BIT(DCACHE, DCACHE_CTRL_REG, DCACHE_WFLUSH);
117  /* Wait until the write flushing process is complete. */
118  while (hw_dcache_is_write_flushed() != true) {}
119 
120  /* If DCACHE retention is enabled we must trigger the toggle-type hardware
121  * write flush bit TWICE so as to force it back to "0".*/
122  if (REG_GETF(CRG_TOP, PMU_CTRL_REG, RETAIN_DCACHE)) {
123  REG_SET_BIT(DCACHE, DCACHE_CTRL_REG, DCACHE_WFLUSH);
124  while (hw_dcache_is_write_flushed() != true) {}
125  }
126 
127  /* Clear the completion indication to be ready for the next invocation */
128  hw_dcache_clear_write_flushed();
129 }

158 __STATIC_FORCEINLINE void hw_dcache_disable(HW_DCACHE_DISABLE_MODE mode)
159 {
160  ASSERT_WARNING(mode < HW_DCACHE_DISABLE_INVALID);
161 
162  switch (mode){
163  case HW_DCACHE_DISABLE_NORMAL:
164  hw_dcache_write_flush();
165  REG_CLR_BIT(DCACHE, DCACHE_CTRL_REG, DCACHE_ENABLE);
166  return;
167  case HW_DCACHE_DISABLE_POWERINGDOWN:
168  hw_dcache_write_flush();
169  while (hw_dcache_wbuffer_is_empty() != true) {}
170  REG_CLR_BIT(DCACHE, DCACHE_CTRL_REG, DCACHE_ENABLE);
171  return;
172  case HW_DCACHE_DISABLE_DEBUG:
173  REG_CLR_BIT(DCACHE, DCACHE_CTRL_REG, DCACHE_ENABLE);
174  return;
175  default:
176  ASSERT_WARNING(0);
177  }
178 }

291 __STATIC_INLINE void hw_dcache_set_cacheable_len(uint32_t len)
292 {
293  /* The assertion focuses only in checking the max value of the cacheable length,
294  * i.e. independently of a previously set base address value */
295  ASSERT_WARNING(len < HW_DCACHE_CACHEABLE_LEN_MAX + 1);
296 
297  /* A write flushing process is imperative to be executed before before setting the
298  * dCache Controller from caching into bypass mode when enabled. This is mandatory
299  * for cache coherency to assure that there are no "dirty" words in the cache RAM */
300  if (hw_dcache_get_cacheable_len() > 0 && len == 0) {
301  hw_dcache_write_flush();
302  }
303 
304  REG_SETF(DCACHE, DCACHE_CTRL_REG, DCACHE_LEN, len);
305 }
