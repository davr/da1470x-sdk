56  * and a size of 131072KB of the target memory (MEMORY_OQSPIC_SIZE) the maximum cacheable length

83 __STATIC_INLINE void hw_cache_enable(void)
84 {
85  REG_SET_BIT(CRG_TOP, SYS_CTRL_REG, CACHERAM_MUX);
86  /* Wait until the CACHERAM_MUX=1 (because of the APB Bridge). */

230 {
