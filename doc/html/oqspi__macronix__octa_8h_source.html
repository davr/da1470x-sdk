94 __UNUSED __RETAINED_CODE static uint8_t oqspi_macronix_octa_read_register(uint8_t opcode, uint32_t address,
95  uint8_t mask, HW_OQSPI_BUS_MODE bus_mode)
96 {
97  __DBG_OQSPI_VOLATILE__ uint8_t reg_val;
98 
99  ASSERT_ERROR((opcode == OQSPI_READ_STATUS_REG_OPCODE && address == 0x00000000) ||
100  (opcode == OQSPI_MACRONIX_OCTA_READ_CFG_REG2_OPCODE) ||
101  (opcode == OQSPI_MACRONIX_OCTA_READ_SECURITY_REG_OPCODE && address == 0x00000000));
102 
103  hw_oqspi_cs_enable();
104 
105  switch (bus_mode) {
106  case HW_OQSPI_BUS_MODE_SINGLE:
107  hw_oqspi_write8(opcode);
108  if (opcode == OQSPI_MACRONIX_OCTA_READ_CFG_REG2_OPCODE) {
109  hw_oqspi_write32(address);
110  }
111  break;
112  case HW_OQSPI_BUS_MODE_OCTA:
113  hw_oqspi_write16(CONVERT_OPCODE_TO_DUAL_BYTE(opcode));
114  hw_oqspi_write32(address);
115  hw_oqspi_write16(0x0000);
116  hw_oqspi_write8(0x00);
117  hw_oqspi_dummy8();
118  break;
119  default:
120  ASSERT_ERROR(0);
121  }
122 
123  reg_val = hw_oqspi_read8();
124  hw_oqspi_cs_disable();
125 
126  return (reg_val & mask);
127 }
