233 __STATIC_INLINE void hw_src_enable(HW_SRC_ID id)
234 {
235 
236 
237  ASSERT_WARNING(id == HW_SRC1 || id == HW_SRC2);
238 
239  /*
240  * The under/overflows that occur due to the reconfiguration can be
241  * ignored, so we disable under/overflow notifications until
242  * (SRC_IN_OK == 1 && SRC_OUT_OK == 1).
243  */
244  HW_SRC_REG_SET_BIT(id, SRC1, CTRL_REG, SRC_IN_FLOWCLR);
245  HW_SRC_REG_SET_BIT(id, SRC1, CTRL_REG, SRC_OUT_FLOWCLR);
246 
247  HW_SRC_REG_SET_BIT(id, SRC1, CTRL_REG, SRC_EN);
248 
249  while (HW_SRC_REG_GETF(id, SRC1, CTRL_REG, SRC_IN_OK) == 0 &&
250  HW_SRC_REG_GETF(id, SRC1, CTRL_REG, SRC_OUT_OK) == 0);
251 
252  HW_SRC_REG_CLR_BIT(id, SRC1, CTRL_REG, SRC_IN_FLOWCLR);
253  HW_SRC_REG_CLR_BIT(id, SRC1, CTRL_REG, SRC_OUT_FLOWCLR);
254 }
