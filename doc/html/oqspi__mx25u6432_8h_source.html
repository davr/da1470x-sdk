126  switch (sys_clk) {

138  ASSERT_WARNING(0);

143 __RETAINED_CODE static bool oqspi_mx25u6432_set_dummy_bytes(uint8_t dummy_bytes)
144 {
145  __DBG_OQSPI_VOLATILE__ uint8_t status_reg;
146  __DBG_OQSPI_VOLATILE__ uint8_t config_reg;
147  __DBG_OQSPI_VOLATILE__ uint8_t verify_config_reg;
148  __DBG_OQSPI_VOLATILE__ OQSPI_MX25U6432_DUMMY_BYTES mx25u6432_dummy_bytes = OQSPI_MX25U6432_DUMMY_BYTES_INVALID;
149 
150  switch (dummy_bytes) {
151  case 1:
152  mx25u6432_dummy_bytes = OQSPI_MX25U6432_DUMMY_BYTES_1;
153  break;
154  case 2:
155  mx25u6432_dummy_bytes = OQSPI_MX25U6432_DUMMY_BYTES_2;
156  break;
157  case 3:
158  mx25u6432_dummy_bytes = OQSPI_MX25U6432_DUMMY_BYTES_3;
159  break;
160  case 4:
161  mx25u6432_dummy_bytes = OQSPI_MX25U6432_DUMMY_BYTES_4;
162  break;
163  default:
164  return false;
165  }

167  oqspi_flash_write_enable(HW_OQSPI_BUS_MODE_SINGLE);
168 
169  status_reg = oqspi_macronix_quad_read_register(OQSPI_READ_STATUS_REG_OPCODE, 0xFF);
170  config_reg = oqspi_macronix_quad_read_register(OQSPI_MACRONIX_QUAD_READ_CONFIG_REG_OPCODE, 0xFF);
171 
172  config_reg = (config_reg & ~OQSPI_MX25U6432_DUMMY_BYTES_MASK) |
173  (mx25u6432_dummy_bytes << OQSPI_MX25U6432_DUMMY_BYTES_POS);
174 
175  oqspi_macronix_quad_write_status_and_config_reg(status_reg, config_reg);
176  while (oqspi_flash_is_busy(HW_OQSPI_BUS_MODE_SINGLE));
177  verify_config_reg = oqspi_macronix_quad_read_register(OQSPI_MACRONIX_QUAD_READ_CONFIG_REG_OPCODE, 0xFF);
178 
179  return (verify_config_reg == config_reg);
