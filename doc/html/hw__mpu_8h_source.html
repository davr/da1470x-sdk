265 __STATIC_FORCEINLINE void hw_mpu_config_region(HW_MPU_REGION_NUM region_num, mpu_region_config *cfg)
266 {
267  ASSERT_ERROR(region_num > HW_MPU_REGION_NONE && region_num <= HW_MPU_REGION_7);
268  if (!cfg) {
269  GLOBAL_INT_DISABLE();
270 
271  ARM_MPU_ClrRegion(region_num);
272 
273  GLOBAL_INT_RESTORE();
274  } else {
275  /* The following assertions check whether the start and end addresses of the region intended
276  * to be protected comply with the 32-byte alignment rule as described in the ARM M33 MPU
277  * documentation. */
278  ASSERT_WARNING((cfg->start_addr & MPU_END_ADDRESS_MASK) == 0);
279  ASSERT_WARNING((cfg->end_addr & MPU_END_ADDRESS_MASK) == 0x1F);
280  /* The following assertion checks whether the value of the shareability is other-than 0x01
281  * which will lead to UNPREDICTABLE behavior according to ARMv8 MPU documentation. */
282  ASSERT_ERROR(cfg->shareability != 0x01);
283 
284  GLOBAL_INT_DISABLE();
285  /* Each of the eight M33 MPU regions is configured via a specific 8 bit set in the 32-bit MAIR0 and
286  * MAIR1 registers. The lower four regions are catered by MAIR0 and the upper four by MAIR1
287  * respectively. Please refer to the ARM M33 MPU documentation for a more detailed description. */
288  ARM_MPU_SetMemAttr(region_num, cfg->attributes);
289  ARM_MPU_SetRegion(region_num,
290  ARM_MPU_RBAR(cfg->start_addr, cfg->shareability, ((cfg->access_permissions >> 1) & 0x01), (cfg->access_permissions & 0x01), cfg->execute_never),
291  ARM_MPU_RLAR(cfg->end_addr, region_num));
292  GLOBAL_INT_RESTORE();
293  }
294 }
