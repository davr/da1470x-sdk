285 __STATIC_INLINE HW_WKUP_TRIGGER hw_wkup_get_trigger(HW_GPIO_PORT port, HW_GPIO_PIN pin)
286 {
287  uint32_t polarity = (*(HW_WKUP_POL_P0_BASE_REG + port) & (0x1 << pin)) >> pin;
288  uint32_t key_enabled = (*(HW_WKUP_SELECT_KEY_P0_BASE_REG + port) & (0x1 << pin)) >> pin;
289  uint32_t gpio_enabled = (*(HW_WKUP_SELECT_GPIO_P0_BASE_REG + port) & (0x1 << pin)) >> pin;
290 
291  if (key_enabled) {
292  return polarity ? HW_WKUP_TRIG_LEVEL_LO_DEB : HW_WKUP_TRIG_LEVEL_HI_DEB;
293  }
294  else if (gpio_enabled) {
295  if ((*(HW_WKUP_SELECT1_GPIO_P0_BASE_REG + port) & (0x1 << pin)) >> pin) {
296  return polarity ? HW_WKUP_TRIG_EDGE_LO : HW_WKUP_TRIG_EDGE_HI;
297  }
298  return polarity ? HW_WKUP_TRIG_LEVEL_LO : HW_WKUP_TRIG_LEVEL_HI;
299  }
300  else {
301  return HW_WKUP_TRIG_DISABLED;
302  }
303 }
